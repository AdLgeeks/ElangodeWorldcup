        function selectWinner(element, team) {
            // Remove selection from all cards
            document.querySelectorAll('.glass-card').forEach(card => {
                card.classList.remove('active-selection');
                const dot = card.querySelector('.selection-dot');
                if (dot) dot.classList.add('opacity-0');
            });

            // Add selection to clicked card
            element.classList.add('active-selection');
            element.querySelector('.selection-dot').classList.remove('opacity-0');
        }

        const slider = document.getElementById('goal-slider');
        const goalVal = document.getElementById('goal-value');
        slider.addEventListener('input', () => {
            goalVal.textContent = slider.value;
        });

        // Simple countdown logic
        let time = 9912; // seconds
        const countdownEl = document.getElementById('countdown');
        setInterval(() => {
            if (time <= 0) return;
            time--;
            const h = Math.floor(time / 3600);
            const m = Math.floor((time % 3600) / 60);
            const s = time % 60;
            countdownEl.textContent = `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
        }, 1000);
    </script>
